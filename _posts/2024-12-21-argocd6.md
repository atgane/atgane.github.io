다시 application controller로 돌아와서 CompareAppState라는 함수를 살펴봅시다. 해당 함수는 원격 저장소와 대상 클러스터 사이의 차이를 찾아 재조정할 대상을 정하는 역할을 했습니다. 이번에는 application controller에서 차이가 발생한 대상을 재조정하는 과정을 탐색해보겠습니다.

---

# processAppRefreshQueueItem의 최하층

지금까지 확인한 코드를 살펴보면 applications의 상태를 가져와 원격 저장소와 비교하고 sync할 대상을 탐색하여 app을 갱신하는 코드까지 확인했습니다. 이제는 갱신된 app을 통하여 어떻게 k8s 리소스가 생성되는지 비교해보도록 하겠습니다.

다시 processAppRefreshQueueItem을 살펴보면 defer로 호출되는 절이 있습니다.

```go
// https://github.com/argoproj/argo-cd/blob/079754c63913522803f7dbe4bded6b6de37f7e34/controller/appcontroller.go#L1541
func (ctrl *ApplicationController) processAppRefreshQueueItem() (processNext bool) {
	
	// ...
	
	defer func() {
		if r := recover(); r != nil {
			log.Errorf("Recovered from panic: %+v\n%s", r, debug.Stack())
		}
		// We want to have app operation update happen after the sync, so there's no race condition
		// and app updates not proceeding. See https://github.com/argoproj/argo-cd/issues/18500.
		// ✅ 이 부분에서 refresh queue -> operation queue로 넘어감
		ctrl.appOperationQueue.AddRateLimited(appKey)
		ctrl.appRefreshQueue.Done(appKey)
	}()
	
	// ...
```

`processAppRefreshQueueItem` 함수가 종료됐을 때 호출되는 부분으로 defer절 내부가 실행됩니다. 이 내부에서 `ctrl.appOperationQueue.AddRateLimited(appKey)` 에서 RefreshQueue → OperationQueue로 동작이 이동합니다. appKey를 넘겨 RefreshQueue에서 처리한 대상을 Operation Queue로 전달합니다. 이때 Operation Queue에 전달된 이벤트는 `processAppOperationItem`이라는 함수가 실행합니다.

```go
// https://github.com/argoproj/argo-cd/blob/079754c63913522803f7dbe4bded6b6de37f7e34/controller/appcontroller.go#L934
func (ctrl *ApplicationController) processAppOperationQueueItem() (processNext bool) {
	// ✅ queue data 가져오기
	appKey, shutdown := ctrl.appOperationQueue.Get()
	
	// ...
	
	// ✅ informer에서 object 조회
	obj, exists, err := ctrl.appInformer.GetIndexer().GetByKey(appKey)
	
	// ...

	// ✅ operation이 nil이 아닐 때
	if app.Operation != nil {
		ctrl.processRequestedAppOperation(app)
		ts.AddCheckpoint("process_requested_app_operation_ms")
	}
	
	// ...
	
}
```

Operation Queue 내부는 appKey로 app 오브젝트를 조회하고 operation이 존재하는 경우 `processRequestedAppOperation` 메서드로 app을 전달합니다.

```go
// https://github.com/argoproj/argo-cd/blob/079754c63913522803f7dbe4bded6b6de37f7e34/controller/appcontroller.go#L1317
func (ctrl *ApplicationController) processRequestedAppOperation(app *appv1.Application) {

	// ...

	if isOperationInProgress(app) {
		// 이미 처리중인 경우 종료가 아니면 retry 로직
	} else {
		
		// app phase를 operation running으로 변경
		state = &appv1.OperationState{Phase: synccommon.OperationRunning, Operation: *app.Operation, StartedAt: metav1.Now()}
		ctrl.setOperationState(app, state)
		// ...
	}

	// 목적지 검증 후
	if err := argo.ValidateDestination(context.Background(), &app.Spec.Destination, ctrl.db); err != nil {
		
		// ...
		
		// ✅ 에러가 없는 경우 SyncAppState 호출
	} else {
		ctrl.appStateManager.SyncAppState(app, state)
	}
	
	// ...
	// app state 재검사
	
}
```

이 부분은 app에 대한 “검증” → “생성” → “생성 이후 살태에 대한 로직 처리”로 진행됩니다. 주요한 함수로 `SyncAppState`라는 함수가 있는데 이 부분에서 실제 k8s 오브젝트의 재조정이 일어납니다.

```go
// https://github.com/argoproj/argo-cd/blob/079754c63913522803f7dbe4bded6b6de37f7e34/controller/sync.go#L90
func (m *appStateManager) SyncAppState(app *v1alpha1.Application, state *v1alpha1.OperationState) {
	// ...

	// ✅ CompareAppState 재호출
	compareResult, err := m.CompareAppState(app, proj, revisions, sources, false, true, syncOp.Manifests, isMultiSourceRevision, rollback)

	// ...

	// ✅ 클러스터 조회
	clst, err := m.db.GetCluster(context.Background(), app.Spec.Destination.Server)
	
	// ...
	
	// ✅ syncCtx 생성
	syncCtx, cleanup, err := sync.NewSyncContext(
		// ...
	)
	
	// ...

	if state.Phase == common.OperationTerminating {
		syncCtx.Terminate()
	} else {
		// ✅ terminating이 아니면 Sync 진행
		syncCtx.Sync()
	}
	
	// ...
}
```

함수 내부는 크게 다음 로직이 실행됩니다.

- CompareAppState 재호출로 CompareResult 추출
- 클러스터 조회
- syncCtx 생성
- syncCtx.Sync 호출

여기서 실제로 sync를 수행하는 함수는 syncCtx의 Sync 메서드입니다. syncContext는 `sync.NewSyncContext` 함수로부터 생성되는데 이 함수는 argocd내부의 코드가 아닙니다. 이 함수는 gitops-engine이라는 라이브러리 코드로부터 호출됩니다. 

gitops-engine은 argocd와 fluxcd가 공동으로 사용하는 gitops 엔진입니다. 통합에 관련된 자세한 사항은 아래 문서를 참고하시기 바랍니다.

https://github.com/argoproj/gitops-engine/blob/master/docs/faq.md

gitops engine의 `Sync`메서드는 다음과 같이 동작합니다.

```go
// https://github.com/argoproj/gitops-engine/blob/54992bf42431e71f71f11647e82105530e56305e/pkg/sync/sync_context.go#L394
func (sc *syncContext) Sync() {
	// ...

	if sc.started() {
		// ...
	} else {
	
		// ...
		
		// ✅ dry run으로 사전 동작 체크
		dryRunTasks := tasks
		if sc.applyOutOfSyncOnly {
			dryRunTasks = sc.filterOutOfSyncTasks(tasks)
		}

		sc.log.WithValues("tasks", dryRunTasks).Info("Tasks (dry-run)")
		if sc.runTasks(dryRunTasks, true) == failed {
			sc.setOperationPhase(common.OperationFailed, "one or more objects failed to apply (dry run)")
			return
		}
	}

	// ...
	// sync 시 hook & phase 설정
	// ...
	
	// ✅ 실제 kubernetes apply 실행
	runState := sc.runTasks(tasks, false)

	// runState에 대한 상태 분기 처리
}
```

`Sync` 는 dry-run으로 생성 테스트를 수행하고 올바른 경우 `runTasks` 메서드를 호출합니다. 실제 kubectl apply와 같은 역할은 runTasks에서 일어납니다. 이 함수 내부는 다음과 같습니다. 여기서는 생성 부분을 주로 다루겠습니다.

```go
// https://github.com/argoproj/gitops-engine/blob/54992bf42431e71f71f11647e82105530e56305e/pkg/sync/sync_context.go#L1155
func (sc *syncContext) runTasks(tasks syncTasks, dryRun bool) runState {
	
	// ...

	// tasks에 대한 분리 작업
	for _, task := range tasks {
		if task.isPrune() {
			pruneTasks = append(pruneTasks, task)
		} else {
			createTasks = append(createTasks, task)
		}
	}
	
	// prune 처리
	// ...

	// delete 처리
	// ...

	var tasksGroup syncTasks

	// ...
	
	if len(tasksGroup) > 0 {
		// ✅ 리소스 생성 처리
		state = sc.processCreateTasks(state, tasksGroup, dryRun)
	}
	
	return state
}

// https://github.com/argoproj/gitops-engine/blob/54992bf42431e71f71f11647e82105530e56305e/pkg/sync/sync_context.go#L1265
func (sc *syncContext) processCreateTasks(state runState, tasks syncTasks, dryRun bool) runState {
	ss := newStateSync(state)
	for _, task := range tasks {
		// dryRun인 경우 스킵
		if dryRun && task.skipDryRun {
			continue
		}
		t := task
		ss.Go(func(state runState) runState {
			
			// ...
			
			// ✅ 고루틴으로 applyObject로 리소스 생성 호출
			result, message := sc.applyObject(t, dryRun, validate)
			
			// ...
			
			return state
		})
	}
	return ss.Wait()
}

// https://github.com/argoproj/gitops-engine/blob/54992bf42431e71f71f11647e82105530e56305e/pkg/sync/sync_context.go#L968
func (sc *syncContext) applyObject(t *syncTask, dryRun, validate bool) (common.ResultCode, string) {
	
	// ...
	
	if shouldReplace { // ✅ replace인 경우
		if t.liveObj != nil { // ✅ 이미 리소스가 있을 때
			// ...
			// crd나 ns kind인 경우: kubectl replace 대신 update resource
			if kube.IsCRD(t.targetObj) || t.targetObj.GetKind() == kubeutil.NamespaceKind {
				
				// ...
				
				_, err = sc.resourceOps.UpdateResource(context.TODO(), update, dryRunStrategy)
				
				// ...
				
			} else {
				// ✅ crd, ns kind가 아니면 replace
				message, err = sc.resourceOps.ReplaceResource(context.TODO(), t.targetObj, dryRunStrategy, force)
			}
		} else {
			// ✅ 리소스 없으면 create
			message, err = sc.resourceOps.CreateResource(context.TODO(), t.targetObj, dryRunStrategy, validate)
		}
	} else { // ✅ replace가 아니면 apply
		message, err = sc.resourceOps.ApplyResource(context.TODO(), t.targetObj, dryRunStrategy, force, validate, serverSideApply, sc.serverSideApplyManager, false)
	}
	
	// ...
	
	// ✅ crd인 경우 crd 대기
	if kube.IsCRD(t.targetObj) && !dryRun {
		crdName := t.targetObj.GetName()
		if err = sc.ensureCRDReady(crdName); err != nil {
			sc.log.Error(err, fmt.Sprintf("failed to ensure that CRD %s is ready", crdName))
		}
	}
	return common.ResultCodeSynced, message
}
```

`runTasks` 내부는 create 뿐만 아니라 prune과 delete 작업을 포함합니다. create관련 작업을 보자면 `processCreateTasks` 를 호출하여 생성 작업을 여러 고루틴으로 처리하여 동시에 `applyObject`를 진행하도록 합니다. 이때 `applyObject`내부에는 이미 존재하는 오브젝트인지, crd인지, 존재하지 않았는지 여부에 대한 리소스 분기 처리를 시도합니다. crd나 ns kind인 경우 리소스 삭제를 방지하기 위해 리소스별 실행하는 명령을 분기하여 처리합니다.

---

지금까지 argocd에서 application을 생성하는 과정을 요약해보면 다음과 같이 동작함을 알 수 있습니다.

1. cli에서 app 생성 요청을 grpc로 전달합니다.
2. api server는 grpc요청을 수신하고 application을 생성합니다.
3. 생성된 application은 controller의 informer를 통해 이벤트로 전달됩니다.
4. refresh queue에 오브젝트가 전달되고 CompareAppState 메서드가 실행됩니다.
5. 내부에서 repo server에 grpc 요청으로 manifest를 조회하고 repo server는 manifest를 구성하여 전달합니다.
6. manifest를 통하여 대상 클러스터에 배포된 내역과 비교하여 재조정할 오브젝트를 생성합니다.
7. operation queue로 재조정할 app에 대한 key를 전달하고 operation queue 내부에서 SyncAppState를 호출합니다.
8. SyncAppState 내부에서 gitops engine의 sync context를 만들고 Sync 메서드를 호출합니다.
9. Sync 메서드 내부에서 동기화할 리소스의 tasks를 만들고 여러 고루틴으로 동기화를 진행합니다.
